## W1-1. BLE `handleIncoming` 超量資料遺失

**檔案**：`Final Project/Core/BluetoothTransport.swift:211-235`

### 現況

```swift
private func handleIncoming(_ chunk: Data) {
    if receiveBuffer.isEmpty && expectedLength == 0 {
        guard chunk.count >= 4 else {
            print("BluetoothTransport: chunk too short for length header")
            return
        }
        let length = Int(chunk[0]) << 24 | Int(chunk[1]) << 16
                   | Int(chunk[2]) << 8  | Int(chunk[3])
        guard length > 0 && length <= 64 * 1024 else {
            print("BluetoothTransport: invalid frame length \(length), discarding")
            return
        }
        expectedLength = length
        receiveBuffer.append(chunk.dropFirst(4))
        startReassemblyTimeout()
    } else {
        receiveBuffer.append(chunk)
    }

    if receiveBuffer.count >= expectedLength {
        let message = Data(receiveBuffer.prefix(expectedLength))
        resetBuffer()                  // ← 把整個 buffer 清掉
        onDataReceived?(message)
    }
}
```

### 問題

當一個 BLE chunk 同時包含「訊息 N 末尾 + 訊息 N+1 開頭」時，`resetBuffer()` 把 N+1 的開頭一起清掉。連續落子或 Indication 排隊時很容易發生，症狀是偶發的 desync alert（其實不是真 desync，只是下一筆訊息的 length header 沒了）。

### 修法

保留超量資料、遞迴處理下一筆。

```swift
private func handleIncoming(_ chunk: Data) {
    if receiveBuffer.isEmpty && expectedLength == 0 {
        guard chunk.count >= 4 else {
            print("BluetoothTransport: chunk too short for length header")
            return
        }
        let length = Int(chunk[0]) << 24 | Int(chunk[1]) << 16
                   | Int(chunk[2]) << 8  | Int(chunk[3])
        guard length > 0 && length <= 64 * 1024 else {
            print("BluetoothTransport: invalid frame length \(length), discarding")
            return
        }
        expectedLength = length
        receiveBuffer.append(chunk.dropFirst(4))
        startReassemblyTimeout()
    } else {
        receiveBuffer.append(chunk)
    }

    // Drain as many complete messages as the buffer holds.
    while receiveBuffer.count >= expectedLength && expectedLength > 0 {
        let message = Data(receiveBuffer.prefix(expectedLength))
        let overflow = Data(receiveBuffer.suffix(from: expectedLength))
        resetBuffer()
        onDataReceived?(message)

        guard !overflow.isEmpty else { break }

        // Need at least 4 bytes to read next length header.
        guard overflow.count >= 4 else {
            // Stash the partial header; next chunk will combine.
            receiveBuffer.append(overflow)
            // expectedLength stays 0; the next handleIncoming() call will
            // re-enter the "header" branch once buffer has ≥ 4 bytes.
            // But buffer.isEmpty check would fail, so handle inline:
            startReassemblyTimeout()
            break
        }

        let nextLen = Int(overflow[0]) << 24 | Int(overflow[1]) << 16
                    | Int(overflow[2]) << 8  | Int(overflow[3])
        guard nextLen > 0 && nextLen <= 64 * 1024 else {
            print("BluetoothTransport: invalid next frame length \(nextLen), discarding")
            break
        }
        expectedLength = nextLen
        receiveBuffer.append(overflow.dropFirst(4))
        startReassemblyTimeout()
    }
}
```

> 說明：迴圈版避免遞迴爆 stack（理論上 64KB / 20bytes ≈ 3000 層）。處理「partial header」時把 1-3 bytes 留在 buffer，但 buffer 非空 / expectedLength=0 是正常 reassembly 邏輯不認得的狀態，所以下一次 handleIncoming 進來會走到 `else { receiveBuffer.append(chunk) }`，buffer 會滾大；此時需要在進入時偵測「buffer.count ≥ 4 且 expectedLength == 0」做 header 解析。改成下面這個更乾淨的寫法：

**改成 buffer-then-parse 結構（推薦最終版）**：

```swift
private func handleIncoming(_ chunk: Data) {
    receiveBuffer.append(chunk)
    parseBuffer()
}

private func parseBuffer() {
    while true {
        if expectedLength == 0 {
            guard receiveBuffer.count >= 4 else {
                if !receiveBuffer.isEmpty { startReassemblyTimeout() }
                return
            }
            let length = Int(receiveBuffer[receiveBuffer.startIndex]) << 24
                       | Int(receiveBuffer[receiveBuffer.startIndex + 1]) << 16
                       | Int(receiveBuffer[receiveBuffer.startIndex + 2]) << 8
                       | Int(receiveBuffer[receiveBuffer.startIndex + 3])
            guard length > 0 && length <= 64 * 1024 else {
                print("BluetoothTransport: invalid frame length \(length), discarding")
                resetBuffer()
                return
            }
            expectedLength = length
            receiveBuffer.removeFirst(4)
            startReassemblyTimeout()
        }

        guard receiveBuffer.count >= expectedLength else {
            return  // 等下一個 chunk
        }

        let message = Data(receiveBuffer.prefix(expectedLength))
        receiveBuffer.removeFirst(expectedLength)
        expectedLength = 0
        reassemblyTimeoutTask?.cancel()
        reassemblyTimeoutTask = nil
        onDataReceived?(message)
        // loop continues — try parsing next message in the same buffer
    }
}
```

`resetBuffer()` 的呼叫點（`disconnect()`、`didSubscribeTo`、`didConnect` 等）保持不變。

## W1-3. `MultipeerManager` 缺 `@MainActor`

**檔案**：`Final Project/Core/MultipeerManager.swift:44-45`

### 現況

```swift
@Observable
class MultipeerManager: NSObject {

    // MARK: Public State
    var connectionState: ConnectionState = .notConnected
    var discoveredPeers: [DiscoveredPeer] = []
    var connectedPeerName: String?
    var isHost: Bool = false
    var connectionMode: ConnectionMode = .wifi
    ...
```

### 問題

雖然 transport 內部所有 callback 都用 `Task { @MainActor in ... }` hop 到主執行緒，但 `MultipeerManager` 自己沒標 `@MainActor`，編譯器無法保證所有 caller 都在 main。如果未來有人在背景 thread 寫測試或從 observer thread 讀 `connectionState`，會有 data race。`GameSessionCoordinator` 是 `@MainActor`，但因為 `MultipeerManager` 沒標，coordinator 呼叫 `multipeerManager.send(...)` 等公開方法時不會被檢查。

### 修法

```swift
@Observable
@MainActor                              // ← 加上
final class MultipeerManager: NSObject {  // ← 順手加 final
    ...
}
```

加上後可能需要的小調整：
- `transport.on*` callbacks 已經 hop 到 MainActor 了，不用再改
- 若有 unit test 直接 `MultipeerManager()` 在 XCTest method 中（XCTest 預設不在 MainActor），需要在 test method 加 `@MainActor` 或包 `await MainActor.run`

`MPCTransport` / `BluetoothTransport` 不用標 `@MainActor`（他們是 NSObject delegate，需要 nonisolated callbacks）。

### 驗收

1. `xcodebuild build` 通過，無 concurrency warning
2. 既有 67 個單元測試全綠（測試檔可能要加 `@MainActor` 標註）
3. 跑一次完整連線流程（host + invite + 落子 + 斷線）正常

### 問題

`UIDevice.current.name` 在 iOS 16+ 雖然有限制（給未取得 Entitlement 的 app 回傳通用名稱），但實測仍會回真名（特別是當 app 取得 Bonjour entitlement 時）。WiFi 大廳內所有人能看到使用者本名 + 機型，是隱私問題。

### 修法

#### 1. 新建 `Final Project/Core/PlayerNameProvider.swift`

```swift
//
//  PlayerNameProvider.swift
//  Final Project
//
//  Centralizes the broadcast name used by both MPC and Bluetooth transports.
//  Uses a user-set nickname (UserDefaults), falling back to a random anonymous
//  identifier so we never leak UIDevice.current.name to other peers.
//

import Foundation

enum PlayerNameProvider {
    private static let defaultsKey = "playerNickname"

    /// User-set nickname (nil if not yet set or explicitly cleared).
    static var savedNickname: String? {
        get {
            let raw = UserDefaults.standard.string(forKey: defaultsKey)?
                .trimmingCharacters(in: .whitespacesAndNewlines)
            return (raw?.isEmpty == false) ? raw : nil
        }
        set {
            let trimmed = newValue?.trimmingCharacters(in: .whitespacesAndNewlines)
            if let trimmed, !trimmed.isEmpty {
                UserDefaults.standard.set(trimmed, forKey: defaultsKey)
            } else {
                UserDefaults.standard.removeObject(forKey: defaultsKey)
            }
        }
    }

    /// The name to broadcast. Always non-empty; uses anonymous fallback if no
    /// nickname is set. Adds a 6-char unique suffix so two players named
    /// "Player" don't collide.
    static func broadcastName() -> String {
        let base = savedNickname ?? "Player"
        let suffix = String(UUID().uuidString.prefix(6))
        return "\(base)｜\(suffix)"
    }

    /// True if user has not yet set a nickname; LobbyView uses this to prompt.
    static var needsOnboarding: Bool {
        savedNickname == nil
    }
}
```

```swift
        case .mid:  return 0.15
        case .high: return 0.20
        }
    }
```

#### `Haptics.swift`（W2-3 會詳細用）

```swift
//
//  Haptics.swift
//  Final Project
//
//  Thin convenience around iOS 17's .sensoryFeedback API. Rather than
//  scattering UIImpactFeedbackGenerator everywhere, view code declares the
//  trigger value and the feedback style, and the system handles debouncing.
//

import SwiftUI

enum HapticEvent: Equatable {
    case selection         // tap on a board cell
    case confirm           // confirm move
    case opponentMove      // opponent placed a piece
    case win               // local user wins or game ends
    case warn              // pass turn / draw
    case connect           // peer connected
    case disconnect        // peer left or desync
}

extension View {
    /// Plays sensory feedback when `trigger` changes. Pass nil to disable.
    @ViewBuilder
    func hapticFeedback(_ event: HapticEvent, trigger: some Equatable) -> some View {
        switch event {
        case .selection:
            self.sensoryFeedback(.selection, trigger: trigger)
        case .confirm:
            self.sensoryFeedback(.impact(weight: .medium), trigger: trigger)
        case .opponentMove:
            self.sensoryFeedback(.impact(weight: .light), trigger: trigger)
        case .win:
            self.sensoryFeedback(.success, trigger: trigger)
        case .warn:
            self.sensoryFeedback(.warning, trigger: trigger)
        case .connect:
            self.sensoryFeedback(.success, trigger: trigger)
        case .disconnect:
            self.sensoryFeedback(.error, trigger: trigger)
        }
    }
}
```

**檔案**：`Final Project/Core/BluetoothTransport.swift:175-186`

### 現況

```swift

    guard let peripheral = connectedPeripheral,
          let char = guestToHostCharRef,
          !guestSendQueue.isEmpty else {
        guestWriteInFlight = false   // ← 即使前一筆還在 in-flight 也設 false
```

### 問題

如果 `drainGuestQueue()` 在前一筆 write 還在 in-flight 時被呼叫且 queue 為空，`guestWriteInFlight` 會被誤設為 false。實務上不一定會出包（因為通常等 `didWriteValueFor` 才再 drain），但程式邏輯不清楚。

### 修法

```swift
private func drainGuestQueue() {
    if guestWriteInFlight { return }   // ← 已有 in-flight，啥都不做
    guard let peripheral = connectedPeripheral,
          let char = guestToHostCharRef,
          !guestSendQueue.isEmpty else { return }   // ← 不再亂改 flag

    let packet = guestSendQueue.removeFirst()
    pendingWriteRetry = packet
    guestWriteInFlight = true
    peripheral.writeValue(packet, for: char, type: .withResponse)
}
```

並確認 `didWriteValueFor`（line 448-463）成功路徑會先設 `guestWriteInFlight = false` 再 drain：

```swift
nonisolated func peripheral(_ peripheral: CBPeripheral, didWriteValueFor characteristic: CBCharacteristic, error: Error?) {
    Task { @MainActor in
        if error == nil {
            self.pendingWriteRetry = nil
            self.guestWriteInFlight = false   // ← 加上
            self.drainGuestQueue()
        } else if let retryData = self.pendingWriteRetry {
            self.pendingWriteRetry = nil
            // guestWriteInFlight 仍 true，因為這次 retry 也算 in-flight
            peripheral.writeValue(retryData, for: characteristic, type: .withResponse)
        } else {
            self.guestWriteInFlight = false
            self.onPeerDisconnected?()
        }
    }
}
```

### 驗收

1. 連續快速送 5 個聊天訊息（透過 BLE）→ 全部抵達且順序正確
2. 模擬 transient write fail（可在 retry 路徑加 print 觀察）→ 重試後成功且 `guestWriteInFlight` 回 false

## W2-6. BLE `.poweredOff` 中途切斷未即時通知

### 問題

### 修法

W1-2 的 `peripheralManagerDidUpdateState` / `centralManagerDidUpdateState` 已會處理任何狀態變化。**但**新增「在連線中（role != .none）才 raise disconnect」的條件，避免冷啟動時誤報：

```swift
// peripheralManagerDidUpdateState 加 case
case .poweredOff:

        self.onPeerDisconnected?()
    }
```

```swift
//
//  SoundManager.swift
//  Final Project
//
//  Lightweight audio playback. Loads short clips lazily; respects a
//  user-toggleable mute setting (UserDefaults).
//

import AVFoundation
import Foundation

@Observable
@MainActor
final class SoundManager {
    static let shared = SoundManager()

    enum Sound: String, CaseIterable {
        case piecePlace   = "piece_place"
        case win          = "win"
        case lose         = "lose"
        case connect      = "connect"
        case disconnect   = "disconnect"
        case chat         = "chat"
    }

    private var players: [Sound: AVAudioPlayer] = [:]

    var isMuted: Bool {
        get { UserDefaults.standard.bool(forKey: "soundMuted") }
        set { UserDefaults.standard.set(newValue, forKey: "soundMuted") }
    }

    private init() {
        try? AVAudioSession.sharedInstance().setCategory(.ambient, mode: .default, options: [.mixWithOthers])
    }

    func play(_ sound: Sound) {
        guard !isMuted else { return }
        let player = players[sound] ?? loadPlayer(for: sound)
        guard let player else { return }
        players[sound] = player
        player.currentTime = 0
        player.play()
    }

    private func loadPlayer(for sound: Sound) -> AVAudioPlayer? {
        guard let url = Bundle.main.url(forResource: sound.rawValue, withExtension: "wav") else {
            print("SoundManager: missing \(sound.rawValue).wav")
            return nil
        }
        return try? AVAudioPlayer(contentsOf: url)
    }
}
```

## W3-3. Canvas 重寫 GomokuBoard（效能）

**檔案**：
- `Final Project/Games/Gomoku/GomokuGameView.swift:63-95`
- 可能新建 `Final Project/Games/Gomoku/GomokuBoardCanvas.swift`

### 現況

25×25 棋盤 = 625 個 `GomokuCellView` (ZStack + Canvas + 多個 overlay)。Pinch zoom 時整個 board content 重繪。

### 問題

低階手機（iPhone SE 3）pinch 可能掉 frame。即使能跑也是浪費資源。

### 修法

把整個棋盤畫進**單一 Canvas**：grid lines、star points、stones 全在 Canvas 內畫，最後用一個透明的 overlay grid 接 tap gesture。

```swift
//
//  GomokuBoardCanvas.swift
//  Final Project
//
//  Replaces the 625-cell ZStack with a single Canvas. ~10× faster on pinch.
//

import SwiftUI

struct GomokuBoardCanvas: View {
    @Bindable var engine: GomokuEngine
    let cellSize: CGFloat

    private var size: Int { engine.rules.boardSize }
    private var totalSize: CGFloat { CGFloat(size) * cellSize }

    var body: some View {
        ZStack {
            Color.gomokuBoard

            Canvas { context, _ in
                drawGrid(context: context)
                drawStars(context: context)
                drawStones(context: context)
                drawLastMoveMarker(context: context)
                drawPendingMove(context: context)
            }
        }
        .frame(width: totalSize, height: totalSize)
        .clipShape(RoundedRectangle(cornerRadius: Radius.s))
        .elevation(.mid)
        .overlay(tapOverlay)   // separate overlay for tap dispatch
    }

    // MARK: Drawing

    private func drawGrid(context: GraphicsContext) {
        for i in 0..<size {
            let p = (CGFloat(i) + 0.5) * cellSize
            // horizontal
            var h = Path()
            h.move(to: CGPoint(x: 0.5 * cellSize, y: p))
            h.addLine(to: CGPoint(x: (CGFloat(size) - 0.5) * cellSize, y: p))
            // vertical
            var v = Path()
            v.move(to: CGPoint(x: p, y: 0.5 * cellSize))
            v.addLine(to: CGPoint(x: p, y: (CGFloat(size) - 0.5) * cellSize))
            context.stroke(h, with: .color(.primary.opacity(0.4)), lineWidth: 0.5)
            context.stroke(v, with: .color(.primary.opacity(0.4)), lineWidth: 0.5)
        }
    }

    private func drawStars(context: GraphicsContext) {
        guard size >= 13 else { return }
        let mid = size / 2
        let off = size >= 19 ? 3 : 2
        let positions: [(Int, Int)] = [
            (off, off), (off, mid), (off, size - 1 - off),
            (mid, off), (mid, mid), (mid, size - 1 - off),
            (size - 1 - off, off), (size - 1 - off, mid), (size - 1 - off, size - 1 - off)
        ]
        for (r, c) in positions {
            let center = CGPoint(x: (CGFloat(c) + 0.5) * cellSize, y: (CGFloat(r) + 0.5) * cellSize)
            context.fill(
                Path(ellipseIn: CGRect(x: center.x - 2.5, y: center.y - 2.5, width: 5, height: 5)),
                with: .color(.primary.opacity(0.5))
            )
        }
    }

    private func drawStones(context: GraphicsContext) {
        let radius = cellSize * 0.42
        for r in 0..<size {
            for c in 0..<size {
                guard engine.board[r][c] != .empty else { continue }
                let isPending = engine.pendingMove?.row == r && engine.pendingMove?.col == c
                guard !isPending else { continue }   // pending drawn separately
                let center = CGPoint(x: (CGFloat(c) + 0.5) * cellSize, y: (CGFloat(r) + 0.5) * cellSize)
                let rect = CGRect(x: center.x - radius, y: center.y - radius, width: radius * 2, height: radius * 2)
                let color: Color = engine.board[r][c] == .black ? .pieceBlack : .pieceWhite
                context.fill(Path(ellipseIn: rect), with: .color(color))
            }
        }
    }

    private func drawLastMoveMarker(context: GraphicsContext) {
        guard let last = engine.lastMove else { return }
        let center = CGPoint(x: (CGFloat(last.col) + 0.5) * cellSize, y: (CGFloat(last.row) + 0.5) * cellSize)
        let r = cellSize * 0.25
        context.stroke(
            Path(ellipseIn: CGRect(x: center.x - r, y: center.y - r, width: r * 2, height: r * 2)),
            with: .color(.red),
            lineWidth: 1.5
        )
    }

    private func drawPendingMove(context: GraphicsContext) {
        guard let pending = engine.pendingMove else { return }
        let center = CGPoint(x: (CGFloat(pending.col) + 0.5) * cellSize, y: (CGFloat(pending.row) + 0.5) * cellSize)
        let radius = cellSize * 0.42
        let rect = CGRect(x: center.x - radius, y: center.y - radius, width: radius * 2, height: radius * 2)
        let baseColor: Color = engine.currentPlayer == .black ? .pieceBlack : .pieceWhite
        context.fill(Path(ellipseIn: rect), with: .color(baseColor.opacity(0.45)))
        context.stroke(Path(ellipseIn: rect), with: .color(.yellow), lineWidth: 2)
    }

    // MARK: Tap dispatch

    private var tapOverlay: some View {
        GeometryReader { _ in
            ForEach(0..<size, id: \.self) { row in
                ForEach(0..<size, id: \.self) { col in
                    Color.clear
                        .contentShape(Rectangle())
                        .frame(width: cellSize, height: cellSize)
                        .position(
                            x: (CGFloat(col) + 0.5) * cellSize,
                            y: (CGFloat(row) + 0.5) * cellSize
                        )
                        .onTapGesture { engine.handleTap(row: row, col: col) }
                }
            }
        }
    }
}
```

## W3-4. MessageEnvelope versioning

**檔案**：
- `Final Project/Core/GameEngine.swift:46-58`
- `Final Project/Core/GameSessionCoordinator.swift:217`（與 W2-8 配合）

### 現況

```swift
struct MessageEnvelope: Codable {
    let type: MessageType
    let gameType: String?
    let payload: Data
    ...
}
```

無版本欄位。schema 升級時 guest 收到新版 payload 直接 decode 失敗（W2-8 加了 alert，但無法區分「對方版本太新」vs「payload 損壞」）。

### 修法

```swift
struct MessageEnvelope: Codable {
    /// Protocol version. Bump when adding new MessageType cases or changing
    /// the wire format of any payload struct. Receivers should:
    ///   - Same major: try to decode; unknown fields are tolerated by Codable
    ///   - Different major: refuse and show "version mismatch" alert
    let version: Int
    let type: MessageType
    let gameType: String?
    let payload: Data

    static let currentVersion = 2

    init(version: Int = MessageEnvelope.currentVersion,
         type: MessageType, gameType: String?, payload: Data) {
        self.version = version
        self.type = type
        self.gameType = gameType
        self.payload = payload
    }

    func encode() -> Data? {
        try? JSONEncoder().encode(self)
    }

    static func decode(from data: Data) -> MessageEnvelope? {
        try? JSONDecoder().decode(MessageEnvelope.self, from: data)
    }
}
```

注意 `version: Int = MessageEnvelope.currentVersion` 預設值讓現有 caller 不用改。`Codable` 對舊版 envelope（沒 version 欄位）會 decode 失敗，所以這個 v0.3.0 版本的 protocol 會與 v0.2.0 完全不相容（兩邊都需要更新到 v0.3.0）。這是合理的單次 break。

#### Coordinator 加版本檢查

```swift

    if envelope.version != MessageEnvelope.currentVersion {
        startGameErrorMessage = "對方使用不同版本的 App（v\(envelope.version) vs v\(MessageEnvelope.currentVersion)），請升級到相同版本後再對戰。"
        showStartGameErrorAlert = true
        multipeerManager.disconnect()
        return
    }
    switch envelope.type {
    ...
    }
}
```

### 驗收

1. 雙方都更新到 v0.3.0 → 一切正常
2. 模擬版本不符（手動把一邊改成 currentVersion = 99）→ alert「對方使用不同版本」
3. v0.3.0 與 v0.2.0 連線：v0.3.0 端會看到 envelope decode failed → desync alert（這是預期行為）

### 修法

把 rematch 相關狀態與行為抽到子物件：

```swift
//
//  RematchVoting.swift
//  Final Project
//
//  Owns the rematch voting state machine. The coordinator delegates all
//  rematch-related UI state and message handling to this object.
//

import SwiftUI

@Observable
@MainActor
final class RematchVoting {
    /// Peer requested a rematch; show accept/reject alert.
    var showVoteAlert: Bool = false

    /// Local user requested a rematch; show waiting overlay.
    var waitingForResponse: Bool = false

    /// Peer rejected the local user's request.
    var rejectedAlert: Bool = false

    /// Called when the local user (engine) requests rematch.
    /// Sends .restartVote envelope. Returns false if a vote is already in flight.
    @discardableResult
    func requestRematch(send: (MessageEnvelope) -> Void, hasEngine: Bool) -> Bool {
        guard !waitingForResponse else { return false }
        guard hasEngine else { return false }
        waitingForResponse = true
        send(MessageEnvelope(type: .restartVote, gameType: nil, payload: Data()))
        return true
    }

    /// Called when the local user taps accept/reject.
    /// Returns true if the engine should reset (accepted).
    func respond(accepted: Bool, send: (MessageEnvelope) -> Void) -> Bool {
        let payload = RestartResponsePayload(accepted: accepted)
        send(MessageEnvelope(type: .restartResponse, gameType: nil, payload: payload.toData()))
        return accepted
    }

    /// Called when an envelope of type .restartVote arrives.
    func receiveVoteRequest() {
        guard !showVoteAlert else { return }
        showVoteAlert = true
    }

    /// Called when an envelope of type .restartResponse arrives.
    /// Returns true if the engine should reset (peer accepted).
    func receiveResponse(_ envelope: MessageEnvelope) -> Bool {
        waitingForResponse = false
        guard let payload = RestartResponsePayload.fromData(envelope.payload) else {
            return false
        }
        if !payload.accepted { rejectedAlert = true }
        return payload.accepted
    }

    /// Called when peer leaves the room mid-vote.
    func peerLeft() {
        waitingForResponse = false
    }

    /// Called when game is reset (between matches).
    func reset() {
        showVoteAlert = false
        waitingForResponse = false
        rejectedAlert = false
    }
}
```

## W3-8. Reassembly Task race 改寫

**檔案**：`Final Project/Core/BluetoothTransport.swift:237-247`

### 現況

```swift
private func startReassemblyTimeout() {
    reassemblyTimeoutTask?.cancel()
    reassemblyTimeoutTask = Task { [weak self] in
        try? await Task.sleep(nanoseconds: 2_000_000_000)
        guard !Task.isCancelled else { return }
        await MainActor.run { [weak self] in
            print("BluetoothTransport: reassembly timeout — resetting buffer")
            self?.resetBuffer()
        }
    }
}
```

### 問題

雖然 `weak self` 保證安全，但 cancel race 在 log 上可能仍打出「reassembly timeout」訊息；且 `Task.isCancelled` 在 sleep 喚醒後檢查可能已過時。

### 修法

```swift
private func startReassemblyTimeout() {
    reassemblyTimeoutTask?.cancel()
    reassemblyTimeoutTask = Task { @MainActor [weak self] in
        try? await Task.sleep(for: .seconds(2))
        guard !Task.isCancelled, let self else { return }
        // Re-check we still have a partial buffer (may have been drained)
        guard !self.receiveBuffer.isEmpty || self.expectedLength > 0 else { return }
        print("BluetoothTransport: reassembly timeout — resetting buffer")
        self.resetBuffer()
    }
}
```

關鍵改動：
- `Task { @MainActor` 直接標 actor，免去 `MainActor.run`
- 用 `Task.sleep(for: .seconds(2))`（modern API）
- 喚醒後再檢查 buffer 狀態才執行 reset

### 驗收

1. 正常傳訊不會看到 timeout log
2. 模擬遺失第二片 chunk → 2 秒後看到 log + buffer 清空
3. 傳訊完成後在 2 秒內呼叫 `disconnect()` → 不會看到誤報 timeout log
